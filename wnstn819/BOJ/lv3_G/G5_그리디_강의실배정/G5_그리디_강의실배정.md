        }

        PriorityQueue<Integer> room = new PriorityQueue<>();
        room.add(0);

        while(!que.isEmpty()){
            int[] list = que.poll();
            if(list[0] >= room.peek()){
                room.poll();
            }
            room.add(list[1]);

        }

        System.out.println(room.size());





    }

}
